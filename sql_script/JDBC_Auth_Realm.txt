--Создание таблиц для авторизации пользователей в GlassFish
CREATE TABLE "USERS" (
    "USER_NAME" VARCHAR(255) NOT NULL,
    "PASSWORD" VARCHAR(255));
Commit;

ALTER TABLE "USERS"
ADD CONSTRAINT "PK_USERS"
PRIMARY KEY ("USER_NAME");
Commit;

CREATE TABLE "GROUPS" (
    "USER_NAME" VARCHAR(255),
    "GROUP_NAME" VARCHAR(255));
Commit;

ALTER TABLE "GROUPS"
ADD CONSTRAINT "GROUPS_USERS_FK1"
FOREIGN KEY ("USER_NAME")
REFERENCES "USERS"("USER_NAME")
USING INDEX "GROUPS_USERS_FK1_INDEX";
Commit;
